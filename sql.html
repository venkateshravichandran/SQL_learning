Interactive PostgreSQL DDL Guide

Database Schema Table Index Constraint

This guide provides an interactive reference for PostgreSQL's Data Definition Language (DDL). Select an object type from the navigation above to see all related commands. Each code block has a copy button for your convenience.

Database

CREATE DATABASE

Creates a new database. A database is the top-level container for all other objects.

-- Create a new database named 'sql_learning'
CREATE DATABASE sql_learning;

ALTER DATABASE ... RENAME TO

Changes the name of an existing database.

-- Rename the database 'my_database' to 'my_database_new'
ALTER DATABASE my_database RENAME TO my_database_new;

DROP DATABASE

Deletes an entire database, including all objects within it. This action is irreversible.

-- Drop the database 'sql_learning'
DROP DATABASE sql_learning;

-- Use IF EXISTS to prevent an error if it doesn't exist
DROP DATABASE IF EXISTS sql_learning;

Schema

CREATE SCHEMA

Creates a new schema, which acts as a namespace or a logical grouping for tables and other objects.

-- Create a new schema named 'my_schema'
CREATE SCHEMA my_schema;

ALTER SCHEMA ... RENAME TO

Changes the name of an existing schema.

-- Rename the schema 'my_schema' to 'car_schema'
ALTER SCHEMA my_schema RENAME TO car_schema;

DROP SCHEMA

Deletes a schema and all objects contained within it.

-- Drop the schema 'schema_name'
DROP SCHEMA schema_name;

Table

CREATE TABLE

Creates a new table to store data in rows and columns.

-- Create a table with three columns and a primary key
CREATE TABLE my_table (
    serial_no SERIAL PRIMARY KEY,
    name VARCHAR(50),
    salary INT
);

ALTER TABLE

Modifies the structure of an existing table, such as adding, dropping, renaming, or modifying columns.

-- Add a new column
ALTER TABLE cars ADD COLUMN price INT;

-- Drop an existing column
ALTER TABLE cars DROP COLUMN price;

-- Modify a column's data type
ALTER TABLE cars ALTER COLUMN years TYPE VARCHAR(4);

-- Rename a column
ALTER TABLE cars RENAME COLUMN year TO years;

ALTER TABLE ... RENAME TO

Changes the name of an existing table.

-- Rename the table 'car' to 'car_table'
ALTER TABLE car RENAME TO car_table;

TRUNCATE TABLE

Quickly removes all rows from a table. This is much faster than DELETE for clearing entire tables.

-- Truncate a single table
TRUNCATE TABLE car;

-- Truncate multiple tables
TRUNCATE TABLE car, temp;

DROP TABLE

Deletes a table and all of its data permanently.

-- Drop the table 'my_table'
DROP TABLE my_table;

-- Use IF EXISTS to prevent an error
DROP TABLE IF EXISTS my_table;

Index

CREATE INDEX

Creates an index on one or more columns of a table to speed up query performance.

-- Create an index on the 'serial_no' column
CREATE INDEX my_index ON my_table (serial_no);

ALTER INDEX ... RENAME TO

Changes the name of an existing index.

-- Rename the index 'my_index' to 'table_index'
ALTER INDEX my_index RENAME TO table_index;

DROP INDEX

Deletes an existing index.

-- Drop the index named 'index_name'
DROP INDEX index_name;

Constraint

ALTER TABLE ... ADD CONSTRAINT

Adds a rule to a table to enforce data integrity. Common types include PRIMARY KEY, FOREIGN KEY, UNIQUE, and CHECK.

-- Add a UNIQUE constraint
ALTER TABLE car ADD CONSTRAINT unique_serial_no UNIQUE(serial_no);

-- Add a CHECK constraint
ALTER TABLE car ADD CONSTRAINT price_check CHECK(price > 500000);

-- Add a PRIMARY KEY constraint
ALTER TABLE car ADD CONSTRAINT pk_car PRIMARY KEY(serial_no);

-- Add a FOREIGN KEY constraint
ALTER TABLE car ADD CONSTRAINT fk_temp FOREIGN KEY(serial_no) REFERENCES temp (serial_no);

ALTER TABLE ... RENAME CONSTRAINT

Changes the name of an existing constraint on a table.

-- Rename 'my_constraint' to 'table_constraint'
ALTER TABLE car RENAME CONSTRAINT my_constraint TO table_constraint;

ALTER TABLE ... DROP CONSTRAINT

Removes a constraint from a table.

-- Drop the constraint named 'primary_key'
ALTER TABLE car DROP CONSTRAINT primary_key;
